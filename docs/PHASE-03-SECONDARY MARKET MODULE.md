    active_listings: number;
    total_sold: number;
    total_rented: number;
    profile_views: number;
    avg_response_time: number;  // in hours
  };
  
  // Reviews
  rating: number;
  review_count: number;
  reviews: Review[];
  
  // Listings
  listings: PropertyListing[];
}

  review_id: string;
  reviewer_name: string;
  rating: number;
  comment: string;
  transaction_type: 'bought' | 'sold' | 'rented';
  date: Date;
  is_verified: boolean;
}
